CREATE DATABASE todoapp;

USE todoapp;

CREATE TABLE tasks(id INT UNSIGNED NOT NULL PRIMARY KEY AUTO_INCREMENT, 
title VARCHAR(255) NOT NULL,
detail TEXT NULL, 
priority INT NOT NULL DEFAULT 0,
period DATETIME NULL,
created_at DATETIME NOT NULL, 
updated_at DATETIME NULL)ENGINE=InnoDB, DEFAULT CHARSET=utf8;
